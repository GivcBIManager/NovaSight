   kubectl exec -it $(kubectl get pod -l app.kubernetes.io/component=backend -n {{ .Release.Namespace }} -o jsonpath='{.items[0].metadata.name}') -n {{ .Release.Namespace }} -- curl -s localhost:5000/api/v1/health
